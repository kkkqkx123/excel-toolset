```typescript
// 2. 解析前置元数据
function parseSkillDocument(markdown) {
  // 提取 --- 包围的YAML前置数据
  const frontmatterMatch = /^---\s*\n([\s\S]*?)\n---\s*\n?/u.exec(markdown);

  if (!frontmatterMatch) return null;

  const frontmatterBlock = frontmatterMatch[1];
  const frontmatter = parseYamlFrontmatter(frontmatterBlock);

  // 提取技能体
  const body = markdown.slice(frontmatterMatch[0].length).trimStart();

  return {
    frontmatter: {
      name: frontmatter.name,
      description: frontmatter.description,
      compatibility: frontmatter.compatibility
    },
    body
  };
}

// 3. 构建技能定义
function buildSkillDefinition(location, markdown, parsed) {
  return {
    name: parsed.frontmatter.name,
    description: parsed.frontmatter.description,
    compatibility: parsed.frontmatter.compatibility,
    location: location,
    sourceKind: 'bundled',
    markdown: markdown,
    body: parsed.body
  };
}
```

```typescript
// 构建技能目录
function buildCatalog(): AgentSkillDefinition[] {
  const definitions: AgentSkillDefinition[] = [];

  // 1. 扫描内建技能
  for (const source of BUNDLED_SKILL_SOURCES) {
    const parsed = parseSkillDocument(source.markdown);
    if (!parsed) continue;

    definitions.push(buildDefinition({
      location: source.location,
      markdown: source.markdown,
      frontmatter: parsed.frontmatter,
      sourceKind: 'bundled',
      body: parsed.body
    }));
  }

  // 2. 按名称排序
  definitions.sort((a, b) => a.name.localeCompare(b.name));

  return definitions;
}
```

```typescript
// 合并内建和外部技能
export function mergeAgentSkillDefinitions(
  preferred: readonly AgentSkillDefinition[],
  fallback: readonly AgentSkillDefinition[]
): AgentSkillDefinition[] {
  const byName = new Map<string, AgentSkillDefinition>();

  // 优先使用preferred技能
  for (const skill of preferred) {
    byName.set(skill.name.toLowerCase(), skill);
  }

  // 补充fallback技能
  for (const skill of fallback) {
    const key = skill.name.toLowerCase();
    if (!byName.has(key)) {
      byName.set(key, skill);
    }
  }

  return Array.from(byName.values())
    .sort((a, b) => a.name.localeCompare(b.name));
}
```

```typescript
// 激活存储结构
interface SkillActivationStore {
  // 全局激活的技能
  globallyEnabled: Set<string>;

  // 会话激活的技能
  sessionEnabled: Set<string>;

  // 禁用的技能
  disabled: Set<string>;
}

// 过滤激活的技能
function filterAgentSkillsByEnabledState(
  skills: readonly AgentSkillDefinition[],
  activations: SkillActivationStore
): AgentSkillDefinition[] {
  return skills.filter(skill => {
    const key = skill.name.toLowerCase();

    // 优先检查禁用
    if (activations.disabled.has(key)) {
      return false;
    }

    // 检查会话激活
    if (activations.sessionEnabled.has(key)) {
      return true;
    }

    // 检查全局激活
    if (activations.globallyEnabled.has(key)) {
      return true;
    }

    // 默认禁用
    return false;
  });
}
```

```typescript
async function handleReadAction(params, readCache) {
  // 1. 查找技能
  const skill = await getAgentSkillByName(params.name);
  if (!skill) {
    return {
      content: [{ type: 'text', text: `Skill not found: ${params.name}` }],
      details: { notFound: true }
    };
  }

  // 2. 检查缓存
  if (!params.refresh && readCache) {
    const cached = readCache.get(skill.name);
    if (cached) {
      return {
        content: [{ type: 'text', text: cached }],
        details: { cached: true }
      };
    }
  }

  // 3. 返回技能内容
  return {
    content: [{ type: 'text', text: skill.body }],
    details: {
      name: skill.name,
      description: skill.description,
      location: skill.location
    }
  };
}
```

```typescript
async function handleInstallAction(params, readCache) {
  // 1. 解析技能文档
  const parsed = parseSkillDocument(params.markdown);
  if (!parsed) {
    return {
      content: [{ type: 'text', text: 'Invalid skill document' }],
      details: { error: 'parse_error' }
    };
  }

  // 2. 构建技能定义
  const skill: AgentSkillDefinition = {
    name: parsed.frontmatter.name,
    description: parsed.frontmatter.description,
    compatibility: parsed.frontmatter.compatibility,
    location: 'external',
    sourceKind: 'external',
    markdown: params.markdown,
    body: parsed.body
  };

  // 3. 保存到外部存储
  await externalStore.saveSkill(skill);

  // 4. 清除缓存
  if (readCache) {
    readCache.invalidate(skill.name);
  }

  // 5. 默认激活
  await enableSkill(skill.name, 'session');

  return {
    content: [{ type: 'text', text: `Skill installed: ${skill.name}` }],
    details: { installed: skill }
  };
}
```

```typescript
interface SkillReadCache {
  get(skillName: string): string | null;
  set(skillName: string, content: string): void;
  invalidate(skillName: string): void;
  clear(): void;
}

// 简单内存缓存实现
class MemorySkillReadCache implements SkillReadCache {
  private cache = new Map<string, string>();

  get(skillName: string): string | null {
    return this.cache.get(skillName) || null;
  }

  set(skillName: string, content: string): void {
    this.cache.set(skillName, content);
  }

  invalidate(skillName: string): void {
    this.cache.delete(skillName);
  }

  clear(): void {
    this.cache.clear();
  }
}
```

```typescript
async function handleAppendAction(params, workbookId) {
  const { scope, content, position } = params;

  // 1. 确定存储键
  const storageKey = scope === 'user'
    ? 'user.instructions'
    : `workbook.instructions.v1.${workbookId}`;

  // 2. 获取现有指令
  const existing = await storage.get(storageKey) || [];
  const instructions = Array.isArray(existing) ? existing : [existing];

  // 3. 插入新指令
  if (position === 'top') {
    instructions.unshift(content);
  } else {
    instructions.push(content);
  }

  // 4. 保存
  await storage.set(storageKey, instructions);

  return {
    content: [{ type: 'text', text: `Instruction appended (${scope})` }],
    details: { count: instructions.length }
  };
}
```

```typescript
function buildSystemPrompt(context: SystemPrompt): string {
  const parts: string[] = [];

  // 1. 系统角色
  parts.push('# System Role');
  parts.push('');
  parts.push('You are Pi, an AI assistant for Excel.');
  parts.push('');
  parts.push('## Capabilities');
  parts.push('');
  for (const cap of context.config.capabilities) {
    parts.push(`- ${cap}`);
  }
  parts.push('');

  // 2. 工具披露
  parts.push(context.tools.descriptions);
  parts.push('');

  // 3. 技能披露
  if (context.skills.available.length > 0) {
    parts.push(context.skills.descriptions);
    parts.push('');
  }

  // 4. 约定信息
  if (context.conventions) {
    parts.push('## Format Conventions');
    parts.push('');
    parts.push(`**Default Font:** ${context.conventions.defaults.fontName} ${context.conventions.defaults.fontSize}`);
    parts.push('');
    parts.push('**Available Format Presets:**');
    for (const [name, preset] of Object.entries(context.conventions.formats)) {
      parts.push(`- ${name}: ${preset.format}`);
    }
    parts.push('');
  }

  // 5. 用户指令
  if (context.instructions) {
    if (context.instructions.global.length > 0) {
      parts.push('## User Instructions (Global)');
      parts.push('');
      context.instructions.global.forEach(inst => {
        parts.push(inst);
        parts.push('');
      });
    }

    if (context.instructions.workbook.length > 0) {
      parts.push('## User Instructions (Workbook)');
      parts.push('');
      context.instructions.workbook.forEach(inst => {
        parts.push(inst);
        parts.push('');
      });
    }
  }

  // 6. 执行模式
  parts.push('## Execution Mode');
  parts.push('');
  parts.push(`Current mode: **${context.config.executionMode}**`);
  if (context.config.executionMode === 'safe') {
    parts.push('');
    parts.push('Mutating operations will require pre-execution confirmation.');
  }
  parts.push('');

  return parts.join('\n');
}
```

```typescript
function calculateContextWindow(selection) {
  const CONTEXT_ROWS = 5;
  const MAX_CONTEXT_COLS = 20;

  // 获取已用范围
  const used = selection.worksheet.getUsedRangeOrNullObject();
  used.load('rowIndex,columnIndex,rowCount,columnCount');
  await context.sync();

  if (used.isNullObject) {
    return {
      address: selection.address,
      isEmpty: true
    };
  }

  // 计算上下文边界
  const usedStartRow = used.rowIndex;
  const usedEndRow = used.rowIndex + used.rowCount - 1;
  const usedStartCol = used.columnIndex;
  const usedEndCol = Math.min(
    used.columnIndex + used.columnCount - 1,
    used.columnIndex + MAX_CONTEXT_COLS - 1
  );

  // 包含选择区域±CONTEXT_ROWS
  const contextStartRow = Math.max(usedStartRow, selection.rowIndex - CONTEXT_ROWS);
  const contextEndRow = Math.min(usedEndRow, selection.rowIndex + selection.rowCount - 1 + CONTEXT_ROWS);

  return {
    address: `${colToLetter(usedStartCol)}${contextStartRow + 1}:${colToLetter(usedEndCol)}${contextEndRow + 1}`
  };
}
```

```typescript
function buildContextText(selection, contextRange) {
  const lines: string[] = [];

  // 选择信息
  lines.push(`**Selection:** ${selection.address}`);
  lines.push(`**Context:** ${contextRange.address}`);
  lines.push('');

  // 数据表格
  lines.push(formatAsMarkdownTable(contextRange.values));

  // 选择区域的公式
  const selFormulas = extractFormulas(selection.formulas);
  if (selFormulas.length > 0) {
    lines.push('');
    lines.push(`**Selected formulas:** ${selFormulas.join(', ')}`);
  }

  // 上下文中的错误
  const errors = findErrors(contextRange.values);
  if (errors.length > 0) {
    lines.push('');
    lines.push(`⚠️ **Errors nearby:** ${errors.map(e => `${e.address}=${e.error}`).join(', ')}`);
  }

  return lines.join('\n');
}
```
